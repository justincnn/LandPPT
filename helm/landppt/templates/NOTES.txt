LandPPT 已部署！

访问方式：
{{- if .Values.ingress.enabled }}
{{- range .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ .host }}/web
{{- end }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc -n {{ .Release.Namespace }} {{ include "landppt.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT/web
{{- else }}
  kubectl port-forward -n {{ .Release.Namespace }} svc/{{ include "landppt.fullname" . }} 8000:{{ .Values.service.port }}
  然后打开 http://localhost:8000/web
{{- end }}
